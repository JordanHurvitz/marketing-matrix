            function switchPage(hash) {

                const url = new URL(hash, window.location.origin);

                const pageId = url.hash.split('?')[0];

                const scrollTargetId = url.searchParams.get('scroll_to');


                pages.forEach(page => {

                    if (`#${page.id}` === pageId) {

                        page.classList.remove('hidden');

                    } else {

                        page.classList.add('hidden');

                    }

                });

                sidebarLinks.forEach(link => {

                    if (link.getAttribute('href') === pageId) {

                        link.classList.add('active');

                    } else {

                        link.classList.remove('active');

                    }

                });

                updateContent();


                if (scrollTargetId) {

                    setTimeout(() => {

                        const targetElement = document.getElementById(scrollTargetId);

                        if (targetElement) {

                            targetElement.scrollIntoView({ behavior: 'smooth', block: 'center' });

                            targetElement.classList.add('ref-highlight');

                            setTimeout(() => targetElement.classList.remove('ref-highlight'), 2000);

                        }

                    }, 100);

                }

            }

            

            sidebarLinks.forEach(link => {

                link.addEventListener('click', (e) => {

                    e.preventDefault();

                    const hash = e.currentTarget.getAttribute('href');

                    if (window.location.hash !== hash) {

                       window.location.hash = hash;

                    } else {

                        // If already on the page, just scroll to top

                        document.querySelector('main').scrollTop = 0;
